$(document).ready(function(){
    $("ul.nav_ul li").hover(
    function(){
        $(this).addClass("hover");
        $("ul:first", this).css("visibility","visible");
    },
    function(event){
        $(this).removeClass("hover");
        $("ul:first", this).css("visibility","hidden");
    })
    $(".nav_ul li ul li:has(ul)").find("a:first").append(" >")
})
